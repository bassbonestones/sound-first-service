import json
import sys
from typing import Any, Dict

# TODO: Adjust imports
from app.db import SessionLocal  # type: ignore
from app.models.core import User  # type: ignore


DEFAULT_SOFT_GATES: Dict[str, Any] = {
 "tonal_complexity_stage": {"comfort": 0, "frontier": 1, "ema": 0.0, "attempts": 0, "frontier_success_count": 0},
 "range_usage_stage": {"comfort": 0, "frontier": 1, "ema": 0.0, "attempts": 0, "frontier_success_count": 0},
 "rhythm_complexity_score": {"comfort": 0.10, "frontier": 0.15, "ema": 0.0, "attempts": 0, "frontier_success_count": 0},
 "interval_velocity_score": {"comfort": 0.10, "frontier": 0.15, "ema": 0.0, "attempts": 0, "frontier_success_count": 0},
 "density_notes_per_second": {"comfort": 0.10, "frontier": 0.15, "ema": 0.0, "attempts": 0, "frontier_success_count": 0},
 "tempo_difficulty_score": {"comfort": 0.10, "frontier": 0.15, "ema": 0.0, "attempts": 0, "frontier_success_count": 0}
}


def _parse_soft_gate_json(raw) -> Dict[str, Any]:
 if raw is None or raw == "":
  return {}
 if isinstance(raw, dict):
  return raw
 try:
  return json.loads(raw)
 except Exception:
  return {}


def _merge_defaults(existing: Dict[str, Any]) -> Dict[str, Any]:
 merged = dict(existing)
 for k, v in DEFAULT_SOFT_GATES.items():
  if k not in merged:
   merged[k] = v
  else:
   # Ensure required keys exist without overwriting existing values
   for sub_k, sub_v in v.items():
    if sub_k not in merged[k]:
     merged[k][sub_k] = sub_v
 return merged


def main() -> None:
 session = SessionLocal()
 try:
  users = session.query(User).all()
  updated = 0

  for u in users:
   existing = _parse_soft_gate_json(getattr(u, "soft_gate_state_json", None))
   merged = _merge_defaults(existing)

   if merged != existing:
    # store as text JSON (works even if column is JSONB)
    setattr(u, "soft_gate_state_json", json.dumps(merged))
    updated += 1

  session.commit()
  print(f"Soft gate init complete. Updated {updated} users out of {len(users)}.")

 finally:
  session.close()


if __name__ == "__main__":
 main()

Why this is enough for v1

You don’t need a “recent history” table because Exponential Moving Average (EMA) gives you recency.

The frontier concept lives per dimension, so you can do:

hard caps (range, required capabilities)

soft caps (complexity scores with frontier buffer)

Later, when this stabilizes, you can normalize into a real table:

user_soft_gate_state(user_id, metric_key, comfort, frontier, ema, attempts, frontier_success_count, updated_at)

But JSON is the fastest way to ship.
